package amog.us;

import com.sun.java.accessibility.util.EventID;

public class Swimmer {


    private String Name;
    private String School;
    private String[] eventID = {};
    private double[] time = {};



    Swimmer(String Name,String School,String[] eventID,double[] time){
        this.Name = Name;
        this.School = School;
        this.eventID = eventID;
        this.time = time;
    }

    static void displayArrayString(String[] array) {
        //Makes brackets to make it look nice
        System.out.print("(");
        //makes sure whole array is printed
        for (int i = 0; i < array.length; i++) {
            //prints array and a space
            System.out.print(array[i] + ", ");

        }
        System.out.println(")");}

        static void displayArrayDouble(double[] array) {
        //Makes brackets to make it look nice
        System.out.print("(");
        //makes sure whole array is printed
        for (int i = 0; i < array.length; i++) {
            //prints array and a space
            System.out.print(array[i] + ", ");

        }
        System.out.println(")");

    }

    public String getName() {

        return Name;
    }

    public String getSchool() {

        return School;
    }

    public String[] getEventID() {

        return eventID;
    }

    public double[] getTime() {

        return time;
    }

    public void setEventID(String[] eventID) {
        this.eventID = eventID;
    }

    public void setName(String name) {

        Name = name;
    }

    public void setSchool(String school) {
        School = school;
    }

    public void setTime(double[] time) {

        this.time = time;
    }
    void printObject(){

        System.out.println("Name:" + Name);
        System.out.println("School:" + School);
        System.out.print("Events:");
        displayArrayString(eventID);
        System.out.print("Times:");
        displayArrayDouble(time);
    }




}
