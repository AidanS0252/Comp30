package amog.us;

public class Tablet extends Laptop{
        private double weight;

    Tablet(String ModelName, double Price, int Ram, double battery, String size, double weight) {
        super(ModelName, Price, Ram, battery, size);
        this.weight = weight;
    }

    //Prints out Data
    void display(){
        String ModelName = getModelName();
        double Price = getPrice();
        int Ram = getRam();
        double battery = getBattery();
        String size = getSize();


        System.out.println("Model " + ModelName + ", Price " + Price + " Dollars , Ram " + Ram + " Gigabytes, " +
                "Battery Life " + battery + " hours, Size " + size + ", Weight " + weight + " kg");

    }




}
