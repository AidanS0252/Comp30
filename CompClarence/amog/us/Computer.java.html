package amog.us;

public class Computer {

    //Name of Model
    private String ModelName = new String();
    //Price of computer
    private double Price;
    //Ram of computer
    private int Ram;
    //Constructer
    Computer(String ModelName, double Price, int Ram){
        this.ModelName = ModelName;
        this.Price = Price;
        this.Ram = Ram;

    }

    public void setModelName(String modelName) {
        ModelName = modelName;
    }

    public void setPrice(int price) {
        Price = price;
    }

    public void setRam(int ram) {
        Ram = ram;
    }

    public double getPrice() {
        return Price;
    }

    public int getRam() {
        return Ram;
    }

    public String getModelName() {
        return ModelName;
    }
    //Prints out Data
    void display(){

        System.out.println("Model " + ModelName + ", Price " + Price + " Dollars , Ram " + Ram + " Gigabytes");

    }


}
