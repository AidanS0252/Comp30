    static int[] swap(int[] array, int number1,int number2){

        //Makes a temporary integer that is equal to first entry
        int Pholder = array[number1];
        //Changes the entries to be the same
        array[number1] = array[number2];
        //Changes the second entry to be the first
        array[number2] = Pholder;

        return array;
    }

    static int[] BubbleSort(int[] unsorted){
        System.out.print("Sorting");
        boolean sorted =false;
        while(!sorted) {

            sorted = true;
            for (int i = 0; i < unsorted.length - 1; i++) {

                if (unsorted[i] > unsorted[i + 1]) {
                    unsorted = swap(unsorted, i, i + 1);
                    sorted =false;
                    System.out.print(".");
                }
            }
        }

       return unsorted;
    }

    static int[] InsertionSort(int[] unsorted) {

        System.out.print("Sorted");

        for (int i = 1; i < unsorted.length; ++i) {
            //sets variable for loops and swaps
            int a = unsorted[i];
           //sets variable for loops and swaps
            int b = i - 1;

            // moves elements greater than b one spot ahead
            while (b >= 0 && unsorted[b] > a) {
                unsorted = swap(unsorted, b + 1, b);
                b = b - 1;

            }
            //Sets b+1 to equal a
            unsorted[b + 1] = a;

        }






        return unsorted;}
