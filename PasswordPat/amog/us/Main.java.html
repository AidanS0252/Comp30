package amog.us;
import java.util.Scanner;
public class Main {

    public static void main(String[] args) {


        //sets password
        var Pass = 1234;
        //makes scanner
        var Password = new Scanner(System.in);
        //gets password input
        System.out.println("Input Password");
        var guess = Password.nextInt();
        //sees if password is correct
        if (guess == Pass){
            System.out.println("Correct");
        }
        //exits if password is wrong
        else{
            System.out.println("Incorrect, Security Informed.");
            System.exit(0);
    }
}}
