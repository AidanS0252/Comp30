    static int Choose(int choiceNum){
        //choiceNum: Number of choices available

        Scanner choices = new Scanner(System.in);
        System.out.println("Input choice as int");
        int answer = Integer.parseInt(choices.nextLine());
        //sends error if choice above highest option
        if(answer > choiceNum){
            System.out.println("Invalid choice, above highest option.");
            System.exit(0);
        }
       //sends error if equal to or lower than 0
        if(answer <= 0){
            System.out.println("Invalid choice, can't be 0 or below.");
            System.exit(0);
        }
        return answer;
    }
