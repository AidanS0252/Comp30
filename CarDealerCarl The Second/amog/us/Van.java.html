package amog.us;

public class Van extends Car{
   String storage;
    Van(String model, double mileage, String brand, double price, double gasMileage, int drive, String storage) {
        super(model, mileage, brand, price, gasMileage, drive);
        this.storage = storage;
    }
    public void PrintObject(){
        String brand = getBrand();
        int drive = getDrive();
        double gasMileage = getGasMileage();
        double mileage = getMileage();
        double price = getPrice();
        String model = getModel();


        System.out.println("Model:" + model);
        System.out.println("Mileage:" + mileage + " miles");
        System.out.println("Brand:" + brand);
        System.out.println("Price:" + price);
        System.out.println("Gas Mileage:" + gasMileage);
        System.out.println(drive + " Wheel Drive");
        System.out.println("Storage Size " + storage);
    }

}
