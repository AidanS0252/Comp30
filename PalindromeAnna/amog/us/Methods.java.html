    static String unify(String Phrase){
        //Phrase: The phrase that is put into lowercase and has spaces removed



        String PhraseLow = Phrase.toLowerCase();

        //makes char[] array for swapping characters
        char[] PalinTest = PhraseLow.toCharArray();

        //String to add letters to
        String same = "";
        //adds letters without spaces
        for (int i = 0; i < PalinTest.length; ++i) {
            //sees if Palin test is not a space
            if (PalinTest[i] != ' '){

                same +=(char)ASNUM;
            }
        }
