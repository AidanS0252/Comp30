    static String CaeserCipher(String EnCrypt, int shift){


        
       
    

        //makes char[] array for swapping characters
        char[] PreEncrypt = EnCrypt.toCharArray();

        //String to add encrypted letters to
        String Encrypted = "";
        //encrypts PreEncrypt
        for (int i = 0; i < PreEncrypt.length; ++i) {
            //sees if PreEncrypt is not space
            if (PreEncrypt[i] != ' '){
               int ASNUM = (int)PreEncrypt[i];
               ASNUM += shift;
               Encrypted +=(char)ASNUM;
            }
        }

        //returns the Encrypted message
        return Encrypted;
    }

    public static String UnCaeserCipher(String EnCrypt, int UnShift){
        //shifts by a negative number in order to reverse encryption
        return  CaeserCipher(EnCrypt,-UnShift);
    }

    //automatically decrypts messages
    public static int FregAnaly(String str){
        
        int[] counts = new int[10000];

        char[] array = str.toCharArray();

        for (int i = 0; i < array.length; i++) {

            int ASNUM = (int) array[i];
            if(ASNUM > 33){
            counts[ASNUM] += 1;}
        }

        int highIndex = -1;
        int highValue = 0;
        for (int i = 0; i < counts.length; i++) {
            if(counts[i] > highValue){
                highIndex = i;
                highValue = counts[i];

            }
        }
        return highIndex -101;
        
    }
