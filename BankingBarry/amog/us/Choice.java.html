package amog.us;



import java.util.ArrayList;
import java.util.Scanner;


public class Choice {

    //uses user input to get a variable
    static int Choose(int choiceNum){
        //choiceNum: Number of choices available

        Scanner choices = new Scanner(System.in);
        System.out.println("Input choice as int");
        int answer = Integer.parseInt(choices.nextLine());
        //sends error if choice above highest option
        if(answer > choiceNum){
            System.out.println("Invalid choice, above highest option.");
            System.exit(0);
        }
        //sends error if equal to or lower than 0
        if(answer <= 0){
            System.out.println("Invalid choice, can't be 0 or below.");
            System.exit(0);
        }
        return answer;
    }

    //uses input to get password as int
    static void Passwords(int Correct){
        //Correct: the correct password


        Boolean Right = false;
        //sets password to Correct
        int Pass = Correct;
        //scanner for inputting password
        Scanner Password = new Scanner(System.in);

        System.out.println("Input Password as int");

        int guess = Integer.parseInt(Password.nextLine());

        //allows user to continue if password is correct
        if (guess == Pass){
            System.out.println("Correct");
            Right = true; }
        //exits program if password is incorrect
        else{
            System.out.println("Incorrect, Security Informed.");
            System.exit(0);
        }


    }
    //adds an Account
    static void addAcc(ArrayList arrA, ArrayList arrS, ArrayList arrB, ArrayList arrP){
        //arrA: Account ArrayList to be added to
        //arrB: Balance ArrayList to be added to
        //arrP: Pin ArrayList to be added to
        //makes temp variables for new entries
        String Tname;
        int Tpin;
        int Tsin;
        double Tbalance;



        //gets values for new Account using user input
        Scanner accountAdd = new Scanner(System.in);
        System.out.println("Please input account name as string.");
        Tname = accountAdd.nextLine();
        System.out.println("Please input PIN number as int.");
        Tpin = Integer.parseInt(accountAdd.nextLine());
        System.out.println("Please input SIN number as int.");
        Tsin = Integer.parseInt(accountAdd.nextLine());
        System.out.println("Please input amount of money to add as double.");
        Tbalance = Double.parseDouble(accountAdd.nextLine());



        //adds entries to ArrayLists to end of arr
        arrA.add(Tname);
        arrS.add(Tsin);

    //changes ArrayLists
    static void changeArr(int num, ArrayList arrA, ArrayList arrS, ArrayList arrB, ArrayList arrP){
        //num: element to be changed
        //arrA: Account to be changed
        //arrS: SIN to be changed
        //arrB: Balance ArrayList to be changed
       //arrP: PIN to be changed
        //makes temp variables for new entries
        String Tname;
        int Tpin;
        int Tsin;
        double Tbalance;



        //gets new values using user input
        Scanner accountAdd = new Scanner(System.in);
        System.out.println("Please input account name as string.");
        Tname = accountAdd.nextLine();
        System.out.println("Please input PIN number as int.");
        Tpin = Integer.parseInt(accountAdd.nextLine());
        System.out.println("Please input SIN number as int.");
        Tsin = Integer.parseInt(accountAdd.nextLine());
        System.out.println("Please input new balance as double.");
        Tbalance = Double.parseDouble(accountAdd.nextLine());

        //changes Account with new values
        arrA.set(num,Tname);
        arrS.set(num,Tsin);

    //removes a object from chosen ArrayList
    static void removeObj(ArrayList arrA, ArrayList arrS, ArrayList arrB, ArrayList arrP, int num){
        //arrA,B,P: ArrayList to be removed from
        //num: element to be removed
        arrA.remove(num);
        arrS.remove(num);
        arrB.remove(num);

    }

    //withdraws from account
    static void balanceChange(ArrayList arr, int num){


        //the new balance
        double newBalance;

        Scanner withdrawInput = new Scanner(System.in);
        System.out.println("Please input new balance as double.");
